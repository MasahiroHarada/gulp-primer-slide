Demo 1. SCSSをCSSに変換する

$ cd ./1_compile-scss

Gulpのインストール


$ npm init --yes
$ npm install --dev gulp gulp-sass dart-sass
  • gulp：Gulpライブラリ
  • gulp-sass：Sassコンパイルを行うGulpプラグイン
  • dart-sass：Sassコンパイラ

gulpfile.js

ビルド手順を記述するファイル


const gulp = require('gulp');
const gulpSass = require('gulp-sass');

// ビルドタスクを関数で表現する
function scss() {
  return gulp.src('src/sass/**/*.scss') // scssファイルを取得
    .pipe(gulpSass()) // scssからcssへの変換処理
    .pipe(gulp.dest('dist/css')); // cssディレクトリに出力
}

exports.dev = scss; // 名前をつけてエクスポート

Gulpにおけるビルドの概要

srcでファイルの内容を取得 → ファイル内容の変換処理（プラグイン）をpipeで繋ぐ → 最終的にdestで出力。

ビルド実行！

$ ./node_modules/.bin/gulp dev

devはタスクをエクスポートした名前。

npmスクリプト


"scripts": {
  "start": "gulp dev"
}

npmスクリプトはnode_modulesディレクトリをコマンド探索パスに含めてくれる。なのでコマンドがスッキリ。

もう一度ビルド実行！

$ npm run start

Demo 2. ファイルの変更を監視する


$ cd ../2_watch-file-changes
$ npm init --yes
$ npm install --dev gulp gulp-sass dart-sass

gulpfile.js


const gulp = require('gulp');
const gulpSass = require('gulp-sass');

function watchFiles(done) {
  // watchの第一引数のファイルが変更されたら
  // 第二引数に渡した関数が実行される。
  gulp.watch('src/sass/**/*.scss', scss);

  // 同期処理の場合はタスクの第一引数に渡ってくる
  // コールバック関数を実行してタスクの完了を通知する。
  done();
}

function scss() {
  return gulp.src('src/sass/**/*.scss')
    .pipe(gulpSass())
    .pipe(gulp.dest('dist'));
}

exports.dev = gulp.series(scss, watchFiles); // タスクを順次実行

ビルド実行！


"scripts": {
"start": "gulp dev"
}
            
$ npm run start

Demo 3. EJSをHTMLに変換する


$ cd ../3_compile-templates
$ npm init --yes
$ npm install --dev gulp gulp-ejs

gulpfile.js


const gulp = require('gulp');
const gulpEjs = require('gulp-ejs');

function ejs() {
  return gulp.src([
    'src/ejs/**/*.ejs',
    '!src/ejs/**/_*.ejs' // 処理除外ファイル
  ])
    .pipe(gulpEjs({}, {}, { ext: '.html' }))
    .pipe(gulp.dest('dist'));
}

exports.dev = ejs;

Demo 4. ローカルサーバーを起動する


$ cd ../4_http-server
$ npm init --yes
$ npm install --dev gulp browser-sync

gulpfile.js


const gulp = require('gulp');
const browserSync = require('browser-sync');

// ローカルWebサーバー作成
const server = browserSync.create();

// サーバー再起動
function reload(done) {
  server.reload();
  done();
}

// サーバー起動
function serve(done) {
  server.init({
    server: {
      baseDir: './public'
    }
  });
  done();
}

function watchFiles(done) {
  gulp.watch('./public/**/*.html', reload);
  done();
}

exports.dev = gulp.series(serve, watchFiles);

ビルド実行！


"scripts": {
"start": "gulp dev"
}
            
$ npm run start

Demo 5. タスクを組み合わせる


$ cd ../5_wiring-up
$ npm init --yes
$ npm install --dev gulp gulp-sass gulp-ejs dart-sass browser-sync

gulpfile.js


const gulp = require('gulp');
const gulpSass = require('gulp-sass');
const gulpEjs = require('gulp-ejs');
const browserSync = require('browser-sync');

const server = browserSync.create();

function reload(done) {
  server.reload();
  done();
}

function serve(done) {
  server.init({
    server: {
      baseDir: './dist'
    }
  });
  done();
}

function scss() {
  return gulp.src('src/sass/**/*.scss')
    .pipe(gulpSass())
    .pipe(gulp.dest('dist/css'));
}

function ejs() {
  return gulp.src([
    'src/ejs/**/*.ejs',
    '!src/ejs/**/_*.ejs'
  ])
    .pipe(gulpEjs({}, {}, { ext: '.html' }))
    .pipe(gulp.dest('dist'));
}

function watchFiles(done) {
  gulp.watch('src/sass/**/*.scss', gulp.series(scss, reload));
  gulp.watch('src/ejs/**/*.ejs', gulp.series(ejs, reload));
  done();
}

exports.dev = gulp.series(
  gulp.pallarel(scss, ejs), // 並列処理
  serve,
  watchFiles
);

ビルド実行！


"scripts": {
"start": "gulp dev"
}
            
$ npm run start

👉 http://localhost:3000

function scss() {
  return gulp
    .src(config.src)
    .pipe(sassVariables({
      $env: process.env.NODE_ENV
    }))
    .pipe(plumber())
    .pipe(sourcemaps.init())
    .pipe(
      gulpSass({
        outputStyle: 'expanded',
        linefeed: 'crlf'
      }).on('error', gulpSass.logError)
    )
    .pipe(autoprefixer())
    .pipe(cleancss({ compatibility: 'ie9' }))
    .pipe(sourcemaps.write('.'))
    .pipe(gulp.dest(config.dest));
}
